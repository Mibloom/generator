16  package org.mybatis.generator.codegen.mybatis3;
17  
18  import java.util.List;
19  
20  import org.mybatis.generator.api.ProgressCallback;
21  import org.mybatis.generator.codegen.AbstractJavaClientGenerator;
22  import org.mybatis.generator.codegen.AbstractJavaGenerator;
23  import org.mybatis.generator.codegen.mybatis3.javamapper.SimpleAnnotatedClientGenerator;
24  import org.mybatis.generator.codegen.mybatis3.javamapper.SimpleJavaClientGenerator;
25  import org.mybatis.generator.codegen.mybatis3.model.SimpleModelGenerator;
26  import org.mybatis.generator.codegen.mybatis3.xmlmapper.SimpleXMLMapperGenerator;
27  import org.mybatis.generator.internal.ObjectFactory;
28  
29  /**
30   * Introspected table implementation for generating simple MyBatis3 artifacts.
31   *
32   * <p>Simple means no "by example" methods, flat model, etc.
33   *
34   * @author Jeff Butler
35   */
36  public class IntrospectedTableMyBatis3SimpleImpl extends IntrospectedTableMyBatis3Impl {
37      public IntrospectedTableMyBatis3SimpleImpl() {
38          super();
39      }
40  
41      @Override
42      protected void calculateXmlMapperGenerator(AbstractJavaClientGenerator javaClientGenerator, List<String> warnings,
43              ProgressCallback progressCallback) {
44          if (javaClientGenerator == null) {
45              if (context.getSqlMapGeneratorConfiguration() != null) {
46                  xmlMapperGenerator = new SimpleXMLMapperGenerator();
47              }
48          } else {
49              xmlMapperGenerator = javaClientGenerator.getMatchedXMLGenerator();
50          }
51  
52          initializeAbstractGenerator(xmlMapperGenerator, warnings, progressCallback);
53      }
54  
55      @Override
56      protected AbstractJavaClientGenerator createJavaClientGenerator() {
57          if (context.getJavaClientGeneratorConfiguration() == null) {
58              return null;
59          }
60  
61          String type = context.getJavaClientGeneratorConfiguration().getConfigurationType();
62  
63          AbstractJavaClientGenerator javaGenerator;
64          if ("XMLMAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
65              javaGenerator = new SimpleJavaClientGenerator(getClientProject());
66          } else if ("ANNOTATEDMAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
67              javaGenerator = new SimpleAnnotatedClientGenerator(getClientProject());
68          } else if ("MAPPER".equalsIgnoreCase(type)) { //$NON-NLS-1$
69              javaGenerator = new SimpleJavaClientGenerator(getClientProject());
70          } else {
71              javaGenerator = (AbstractJavaClientGenerator) ObjectFactory.createInternalObject(type);
72          }
73  
74          return javaGenerator;
75      }
76  
77      @Override
78      protected void calculateJavaModelGenerators(List<String> warnings, ProgressCallback progressCallback) {
79  
80          AbstractJavaGenerator javaGenerator = new SimpleModelGenerator(getModelProject());
81          initializeAbstractGenerator(javaGenerator, warnings, progressCallback);
82          javaGenerators.add(javaGenerator);
83      }
84  }
