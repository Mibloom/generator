40  public class RenderingUtilities {
41      private RenderingUtilities() {
42      }
43  
44      public static final String JAVA_INDENT = "    "; //$NON-NLS-1$
45      private static final TypeParameterRenderer typeParameterRenderer = new TypeParameterRenderer();
46      private static final FieldRenderer fieldRenderer = new FieldRenderer();
47      private static final InitializationBlockRenderer initializationBlockRenderer = new InitializationBlockRenderer();
48      private static final MethodRenderer methodRenderer = new MethodRenderer();
49      private static final InnerClassRenderer innerClassRenderer = new InnerClassRenderer();
50      private static final InnerInterfaceRenderer innerInterfaceRenderer = new InnerInterfaceRenderer();
51      private static final InnerEnumRenderer innerEnumRenderer = new InnerEnumRenderer();
52  
53      // should return an empty string if no type parameters
54      public static String renderTypeParameters(List<TypeParameter> typeParameters, CompilationUnit compilationUnit) {
55          return typeParameters.stream()
56                  .map(tp -> typeParameterRenderer.render(tp, compilationUnit))
57                  .collect(CustomCollectors.joining(", ", "<", "> ")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
58      }
59  
60      public static List<String> renderFields(List<Field> fields, CompilationUnit compilationUnit) {
61          return fields.stream()
62                  .flatMap(f -> renderField(f, compilationUnit))
63                  .toList();
64      }
65  
66      private static Stream<String> renderField(Field field, CompilationUnit compilationUnit) {
67          return addEmptyLine(fieldRenderer.render(field, compilationUnit).stream()
68                  .map(RenderingUtilities::javaIndent));
69      }
70  
71      public static List<String> renderInitializationBlocks(List<InitializationBlock> initializationBlocks) {
72          return initializationBlocks.stream()
73                  .flatMap(RenderingUtilities::renderInitializationBlock)
74                  .toList();
75      }
76  
77      private static Stream<String> renderInitializationBlock(InitializationBlock initializationBlock) {
78          return addEmptyLine(initializationBlockRenderer.render(initializationBlock).stream()
79                  .map(RenderingUtilities::javaIndent));
80      }
81  
82      public static List<String> renderClassOrEnumMethods(List<Method> methods, CompilationUnit compilationUnit) {
83          return methods.stream()
84                  .flatMap(m -> renderMethod(m, false, compilationUnit))
85                  .toList();
86      }
87  
88      public static List<String> renderInterfaceMethods(List<Method> methods, CompilationUnit compilationUnit) {
89          return methods.stream()
90                  .flatMap(m -> renderMethod(m, true, compilationUnit))
91                  .toList();
92      }
93  
94      private static Stream<String> renderMethod(Method method, boolean inInterface, CompilationUnit compilationUnit) {
95          return addEmptyLine(methodRenderer.render(method, inInterface, compilationUnit).stream()
96                  .map(RenderingUtilities::javaIndent));
97      }
98  
99      private static Stream<String> addEmptyLine(Stream<String> in) {
100         return Stream.of(in, Stream.of("")) //$NON-NLS-1$
101                 .flatMap(Function.identity());
102     }
103 
104     public static List<String> renderInnerClasses(List<InnerClass> innerClasses, CompilationUnit compilationUnit) {
105         return innerClasses.stream()
106                 .flatMap(ic -> renderInnerClass(ic, compilationUnit))
107                 .toList();
108     }
109 
110     public static List<String> renderInnerClassNoIndent(InnerClass innerClass, CompilationUnit compilationUnit) {
111         return innerClassRenderer.render(innerClass, compilationUnit);
112     }
113 
114     private static Stream<String> renderInnerClass(InnerClass innerClass, CompilationUnit compilationUnit) {
115         return addEmptyLine(innerClassRenderer.render(innerClass, compilationUnit).stream()
116                 .map(RenderingUtilities::javaIndent));
117     }
118 
119     public static List<String> renderInnerInterfaces(List<InnerInterface> innerInterfaces,
120             CompilationUnit compilationUnit) {
121         return innerInterfaces.stream()
122                 .flatMap(ii -> renderInnerInterface(ii, compilationUnit))
123                 .toList();
124     }
125 
126     public static List<String> renderInnerInterfaceNoIndent(InnerInterface innerInterface,
127             CompilationUnit compilationUnit) {
128         return innerInterfaceRenderer.render(innerInterface, compilationUnit);
129     }
130 
131     private static Stream<String> renderInnerInterface(InnerInterface innerInterface, CompilationUnit compilationUnit) {
132         return addEmptyLine(innerInterfaceRenderer.render(innerInterface, compilationUnit).stream()
133                 .map(RenderingUtilities::javaIndent));
134     }
135 
136     public static List<String> renderInnerEnums(List<InnerEnum> innerEnums, CompilationUnit compilationUnit) {
137         return innerEnums.stream()
138                 .flatMap(ie -> renderInnerEnum(ie, compilationUnit))
139                 .toList();
140     }
141 
142     public static List<String> renderInnerEnumNoIndent(InnerEnum innerEnum, CompilationUnit compilationUnit) {
143         return innerEnumRenderer.render(innerEnum, compilationUnit);
144     }
145 
146     private static Stream<String> renderInnerEnum(InnerEnum innerEnum, CompilationUnit compilationUnit) {
147         return addEmptyLine(innerEnumRenderer.render(innerEnum, compilationUnit).stream()
148                 .map(RenderingUtilities::javaIndent));
149     }
150 
151     public static List<String> renderPackage(CompilationUnit compilationUnit) {
152         List<String> answer = new ArrayList<>();
153 
154         String pack = compilationUnit.getType().getPackageName();
155         if (stringHasValue(pack)) {
156             answer.add("package " + pack + ";"); //$NON-NLS-1$ //$NON-NLS-2$
157             answer.add(""); //$NON-NLS-1$
158         }
159         return answer;
160     }
161 
162     public static List<String> renderStaticImports(CompilationUnit compilationUnit) {
163         if (compilationUnit.getStaticImports().isEmpty()) {
164             return Collections.emptyList();
165         }
166 
167         return addEmptyLine(compilationUnit.getStaticImports().stream()
168                 .map(s -> "import static " + s + ";")) //$NON-NLS-1$ //$NON-NLS-2$
169                 .collect(Collectors.toList());
170     }
171 
172     public static List<String> renderImports(CompilationUnit compilationUnit) {
173         Set<String> imports = renderImports(compilationUnit.getImportedTypes());
174 
175         if (imports.isEmpty()) {
176             return Collections.emptyList();
177         }
178 
179         return addEmptyLine(imports.stream()).toList();
180     }
181 
182     private static Set<String> renderImports(Set<FullyQualifiedJavaType> imports) {
183         return imports.stream()
184                 .map(FullyQualifiedJavaType::getImportList)
185                 .flatMap(List::stream)
186                 .map(RenderingUtilities::toFullImport)
187                 .collect(Collectors.toCollection(TreeSet::new));
188     }
189 
190     private static String toFullImport(String s) {
191         return "import " + s + ";"; //$NON-NLS-1$ //$NON-NLS-2$
192     }
193 
194 
195     private static String javaIndent(String in) {
196         if (in.isEmpty()) {
197             return in; // don't indent empty lines
198         }
199 
200         return JAVA_INDENT + in;
201     }
202 
203     public static List<String> removeLastEmptyLine(List<String> lines) {
204         if (lines.get(lines.size() - 1).isEmpty()) {
205             return lines.subList(0, lines.size() - 1);
206         } else {
207             return lines;
208         }
209     }
