16  package org.mybatis.generator.api.dom.kotlin;
17  
18  /**
19   * Nearly complete list of Kotlin modifiers. Modifiers are in the order suggested by Kotlin
20   * coding conventions for reference, but this order is not enforced by the library. Users must add modifiers
21   * in the correct order.
22   */
23  public enum KotlinModifier {
24  
25      PUBLIC("public"), //$NON-NLS-1$
26      PROTECTED("protected"), //$NON-NLS-1$
27      PRIVATE("private"), //$NON-NLS-1$
28      INTERNAL("internal"), //$NON-NLS-1$
29      EXPECT("expect"), //$NON-NLS-1$
30      ACTUAL("actual"), //$NON-NLS-1$
31      FINAL("final"), //$NON-NLS-1$
32      OPEN("open"), //$NON-NLS-1$
33      ABSTRACT("abstract"), //$NON-NLS-1$
34      SEALED("sealed"), //$NON-NLS-1$
35      CONST("const"), //$NON-NLS-1$
36      EXTERNAL("external"), //$NON-NLS-1$
37      OVERRIDE("override"), //$NON-NLS-1$
38      LATE_INIT("lateinit"), //$NON-NLS-1$
39      TAILREC("tailrec"), //$NON-NLS-1$
40      VARARG("vararg"), //$NON-NLS-1$
41      SUSPEND("suspend"), //$NON-NLS-1$
42      INNER("inner"), //$NON-NLS-1$
43      INFIX("infix"), //$NON-NLS-1$
44      OPERATOR("operator"), //$NON-NLS-1$
45      DATA("data"); //$NON-NLS-1$
46  
47      private final String value;
48  
49      KotlinModifier(String value) {
50          this.value = value;
51      }
52  
53      public String getValue() {
54          return value;
55      }
56  }
