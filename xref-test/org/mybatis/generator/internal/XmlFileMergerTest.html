37   * This test is related to issue #87 where XML files are slightly different

44      @Test
45      void testThatFilesAreTheSameAfterMerge() throws Exception {
46          DefaultXmlFormatter xmlFormatter = new DefaultXmlFormatter();
47          Properties p = new Properties();
48          p.setProperty(PropertyRegistry.COMMENT_GENERATOR_SUPPRESS_DATE, "true");
49          CommentGenerator commentGenerator = new DefaultCommentGenerator();
50          commentGenerator.addConfigurationProperties(p);
51  
52          Document document = new Document(XmlConstants.MYBATIS3_MAPPER_PUBLIC_ID,
53                  XmlConstants.MYBATIS3_MAPPER_SYSTEM_ID);
54          document.setRootElement(getSqlMapElement(commentGenerator));
55  
56          GeneratedXmlFile generatedFile1 = new GeneratedXmlFile(document, "TestMapper.xml", "org.mybatis.test", "src",
57                  true, xmlFormatter);
58          InputSource is1 = new InputSource(new StringReader(generatedFile1.getFormattedContent()));
59  
60          GeneratedXmlFile generatedFile2 = new GeneratedXmlFile(document, "TestMapper.xml", "org.mybatis.test", "src",
61                  true, xmlFormatter);
62          InputSource is2 = new InputSource(new StringReader(generatedFile2.getFormattedContent()));
63  
64          String mergedSource = XmlFileMergerJaxp.getMergedSource(is1, is2, "TestMapper.xml");
65  
66          assertEquals(generatedFile1.getFormattedContent(), mergedSource);
67      }

151     private XmlElement getSqlMapElement(CommentGenerator commentGenerator) {
152 
153         XmlElement answer = new XmlElement("mapper");
154         String namespace = "org.mybatis.test.TestMapper";
155         answer.addAttribute(new Attribute("namespace", namespace));
156 
157         commentGenerator.addRootComment(answer);
158 
159         addInsertElement(commentGenerator, answer);
160         addCdataNode1(commentGenerator, answer);
161         addCdataNode2(commentGenerator, answer);
162 
163         return answer;
164     }
165 
166     private void addInsertElement(CommentGenerator commentGenerator, XmlElement parentElement) {
167         XmlElement answer = new XmlElement("insert");
168 
169         answer.addAttribute(new Attribute("id", "insert"));
170 
171         FullyQualifiedJavaType parameterType;
172         parameterType = new FullyQualifiedJavaType("org.mybatis.test.TestRecord");
173 
174         answer.addAttribute(new Attribute("parameterType", parameterType.getFullyQualifiedName()));
175 
176         commentGenerator.addComment(answer);
177 
178         String insertClause = "insert into " +
179                 "myschema.mytable" +
180                 " (id, description)";
181 
182         String valuesClause = "values (#{id}, #{description})";
183 
184         answer.addElement(new TextElement(insertClause));
185 
186         answer.addElement(new TextElement(valuesClause));
187 
188         parentElement.addElement(answer);
189     }
190 
191     private void addCdataNode1(CommentGenerator commentGenerator, XmlElement parentElement) {
192         XmlElement answer = new XmlElement("select");
193         answer.addAttribute(new Attribute("id", "selectWithCdata1"));
194         commentGenerator.addComment(answer);
195 
196         answer.addElement(new TextElement("<![CDATA["));
197         answer.addElement(new TextElement("select foo from bar where foo < 22"));
198         answer.addElement(new TextElement("]]>"));
199 
200         parentElement.addElement(answer);
201     }
202 
203     private void addCdataNode2(CommentGenerator commentGenerator, XmlElement parentElement) {
204         XmlElement answer = new XmlElement("select");
205         answer.addAttribute(new Attribute("id", "selectWithCdata2"));
206         commentGenerator.addComment(answer);
207 
208         answer.addElement(new TextElement("select foo from bar where foo <![CDATA[ < ]]> 22"));
209 
210         parentElement.addElement(answer);
211     }
