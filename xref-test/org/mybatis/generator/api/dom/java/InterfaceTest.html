16  package org.mybatis.generator.api.dom.java;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  import static org.junit.jupiter.api.Assertions.*;
20  
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import org.junit.jupiter.api.Test;
25  import org.mybatis.generator.api.dom.java.render.TopLevelInterfaceRenderer;
26  
27  class InterfaceTest {
28  
29      @Test
30      void testConstructor() {
31          Interface interfaze = new Interface("com.foo.UserInterface");
32          assertNotNull(interfaze);
33      }
34  
35      @Test
36      void testAddImportedType() {
37          Interface interfaze = new Interface("com.foo.UserInterface");
38          FullyQualifiedJavaType arrayList = FullyQualifiedJavaType.getNewArrayListInstance();
39          interfaze.addImportedType(arrayList);
40  
41          assertNotNull(interfaze.getImportedTypes());
42          assertEquals(1, interfaze.getImportedTypes().size());
43          assertTrue(interfaze.getImportedTypes().contains(arrayList));
44      }
45  
46      @Test
47      void testAddImportedTypes() {
48          Interface interfaze = new Interface("com.foo.UserInterface");
49          Set<FullyQualifiedJavaType> importedTypes = new HashSet<>();
50  
51          FullyQualifiedJavaType arrayList = FullyQualifiedJavaType.getNewArrayListInstance();
52          FullyQualifiedJavaType hashMap = FullyQualifiedJavaType.getNewHashMapInstance();
53  
54          importedTypes.add(arrayList);
55          importedTypes.add(hashMap);
56          interfaze.addImportedTypes(importedTypes);
57  
58          assertNotNull(interfaze.getImportedTypes());
59          assertEquals(2, interfaze.getImportedTypes().size());
60          assertTrue(interfaze.getImportedTypes().contains(arrayList));
61          assertTrue(interfaze.getImportedTypes().contains(hashMap));
62      }
63  
64      @Test
65      void testAddFileCommentLine() {
66          Interface interfaze = new Interface("com.foo.UserInterface");
67          interfaze.addFileCommentLine("test");
68  
69          assertNotNull(interfaze.getFileCommentLines());
70          assertEquals(1, interfaze.getFileCommentLines().size());
71          assertEquals("test", interfaze.getFileCommentLines().get(0));
72      }
73  
74      @Test
75      void testAddStaticImport() {
76          Interface interfaze = new Interface("com.foo.UserInterface");
77          interfaze.addStaticImport("com.foo.StaticUtil");
78  
79          assertNotNull(interfaze.getStaticImports());
80          assertEquals(1, interfaze.getStaticImports().size());
81          assertTrue(interfaze.getStaticImports().contains("com.foo.StaticUtil"));
82      }
83  
84      @Test
85      void testAddStaticImports() {
86          Interface interfaze = new Interface("com.foo.UserInterface");
87          Set<String> staticImports = new HashSet<>();
88          staticImports.add("com.foo.StaticUtil1");
89          staticImports.add("com.foo.StaticUtil2");
90          interfaze.addStaticImports(staticImports);
91  
92          assertNotNull(interfaze.getStaticImports());
93          assertEquals(2, interfaze.getStaticImports().size());
94          assertTrue(interfaze.getStaticImports().contains("com.foo.StaticUtil1"));
95          assertTrue(interfaze.getStaticImports().contains("com.foo.StaticUtil2"));
96      }
97  
98      @Test
99      void testInterfaceFields() {
100         Interface interfaze = new Interface("foo.Bar");
101         interfaze.setVisibility(JavaVisibility.PUBLIC);
102 
103         Field field = new Field("EMPTY_STRING", FullyQualifiedJavaType.getStringInstance());
104         field.setInitializationString("\"\"");
105         interfaze.addField(field);
106 
107         field = new Field("ONE", FullyQualifiedJavaType.getStringInstance());
108         field.setInitializationString("\"one\"");
109         interfaze.addField(field);
110 
111         String expected = "package foo;" + System.lineSeparator()
112             + System.lineSeparator()
113             + "public interface Bar {" + System.lineSeparator()
114             + "    String EMPTY_STRING = \"\";" + System.lineSeparator() + System.lineSeparator()
115             + "    String ONE = \"one\";" + System.lineSeparator()
116             + "}";
117 
118         TopLevelInterfaceRenderer renderer = new TopLevelInterfaceRenderer();
119         assertThat(renderer.render(interfaze)).isEqualTo(expected);
120     }
121 }
