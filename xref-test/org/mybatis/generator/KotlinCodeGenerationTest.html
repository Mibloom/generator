18  import static org.junit.jupiter.api.Assertions.assertTrue;
19  
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.junit.jupiter.params.ParameterizedTest;
24  import org.junit.jupiter.params.provider.MethodSource;
25  import org.mybatis.generator.api.GeneratedKotlinFile;
26  import org.mybatis.generator.api.MyBatisGenerator;
27  import org.mybatis.generator.config.Configuration;
28  import org.mybatis.generator.config.xml.ConfigurationParser;
29  import org.mybatis.generator.internal.DefaultShellCallback;
30  
31  /**
32   * This test executes the same generator configuration that is in the mybatis-generator-systests-kotlin
33   * project. The purpose of this test is to exercise the Kotlin code generators during the initial build
34   * as a kind of smoke test - currently we do not try to parse the generated code. This test also shows
35   * test coverage as it mimics the code paths executed by the integration test.
36   *
37   * @author Jeff Butler
38   */
39  class KotlinCodeGenerationTest {
40  
41      @ParameterizedTest
42      @MethodSource("kotlinFileGenerator")
43      void testKotlinParse(GeneratedKotlinFile generatedKotlinFile) {
44          // for now, just let the test pass. if we find a good Kotlin parser, then mimic the
45          // function of the Java test by trying to parse generated code
46          assertTrue(true);
47      }
48  
49      static List<GeneratedKotlinFile> kotlinFileGenerator() throws Exception {
50          JavaCodeGenerationTest.createDatabase();
51          return generateKotlinFiles("/scripts/generatorConfig-kotlin.xml");
52      }
53  
54      static List<GeneratedKotlinFile> generateKotlinFiles(String configFile) throws Exception {
55          List<String> warnings = new ArrayList<>();
56          ConfigurationParser cp = new ConfigurationParser(warnings);
57          Configuration config = cp.parseConfiguration(KotlinCodeGenerationTest.class.getResourceAsStream(configFile));
58  
59          DefaultShellCallback shellCallback = new DefaultShellCallback(true);
60  
61          MyBatisGenerator myBatisGenerator = new MyBatisGenerator(config, shellCallback, warnings);
62          myBatisGenerator.generate(null, null, null, false);
63          return myBatisGenerator.getGeneratedKotlinFiles();
64      }
65  }
